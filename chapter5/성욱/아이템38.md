# any 타입은 가능한 좁은 범위에서만 사용하기

```typescript
function processBar(b: Bar) { /* ... */ }

function f() {
	const x = expressionReturningFoo();
	processBar(x); // 'Foo' 형식의 인수는 'Bar 형식의 매개변수에 할당될 수 없습니다.
}
```

개선 후:

```typescript
// 첫번째 옵션 (비추천)
function f1() {
	const x: any = expressionReturningFoo();
	processBar(x);
	return x; // any 타입
}

// 두번째 옵션 (추천)
function f2() {
	const x = expressionReturningFoo();
	processBar(x as any); // 매개변수만 any로 취급 이후 코드에서는 원래 타입 유지
	return x; // 기존 타입
}

// 세번째 옵션
function f3() {
	const x = expressionReturningFoo();
	// @ts-ignore
	processBar(x); // 이 줄만 타입 체커가 무시됨
	return x;
}
```

객체의 타입 선언

```typescript
// 첫번째 옵션 (비추천)
const config: Config = {
	a: 1,
	b: 2,
	c: {
		key: value // 타입 에러 나는 지점
	}
} as any; // 객체 전체가 any 타입

// 두번째 옵션 (추천)
const config: Config = {
	a: 1,
	b: 2,
	c: {
		key: value as any; // value 값만 any 타입
	}
}
```
