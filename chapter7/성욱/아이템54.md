# 객체를 순회하는 노하우

```typescript
const obj = {
	one: 'uno',
	two: 'dos',
	three: 'tres',
};

for (const k in obj) { // k의 타입이 string이기에 발생하는 오류 (실행은 정상)
	const v = obj[k]; // obj에 인덱스 시그니처가 없기 때문에 엘리먼트는 암시적으로 'any' 타입입니다.
}

// 구체적으로 타입 명시
let k: keyof typeof obj;
for (k in obj) {
	const v = obj[k]; // 정상
}

// 더 나은 방법
function foo(abc: ABC) {
	for (const [k, v] of Object.entries(abc)) {
		k // string
		v // any
	}
}

Object.prototype.z = 3 // 이렇게 사용하면 객체의 정의에 없는 속성이 등장할 수 있다 (오염되니 절대 사용하지 말 것)
```
